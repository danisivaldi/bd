-- Listagem do tipo, modelo e quantidade dos equipamentos de cada atração musical
SELECT A.NOME_ARTISTICO AS BANDA, E.MODELO AS EQUIPAMENTO, E.TIPO, E.QUANTIDADE
    FROM EQUIPAMENTOS E, APRESENTACOES_EQUIPAMENTOS AE, APRESENTACOES A, ATRACOES_MUSICAIS AM 
    WHERE (E.TIPO = AE.TIP_EQUIP AND E.MODELO = AE.MOD_EQUIP) AND (A.ID = AE.APRESENTACAO) AND (A.NOME_ARTISTICO = AM.NOME_ARTISTICO);

-- Listagem do modelo e do fabricante dos fogos de artificio, agrupado por seus fabricantes e ordenado pelo preço
SELECT F.MODELO, F.FABRICANTE, F.PRECO
    FROM FOGOS_ARTIFICIOS F JOIN ARTIFICIOS_SHOW A
    ON A.MOD_FOGOS = F.MODELO AND A.FAB_FOGOS = F.FABRICANTE
        GROUP BY F.FABRICANTE, F.MODELO, F.PRECO
        ORDER BY F.PRECO;
