CREATE TABLE CLIENTES (
    CPF VARCHAR2(14),
    NOME VARCHAR2(20) NOT NULL,
    CONTATO VARCHAR2(13) NOT NULL,
    
    CONSTRAINT PK_CLIENTES PRIMARY KEY (CPF),
    --CONSTRAINT CHECK_CONTATO CHECK(REGEXP_LIKE(CONTATO, ‘[0-9]{2}\-[0-9]{5}\-[0-9]{4}’)),
    CONSTRAINT CHECK_CPF_CLIENTES CHECK(REGEXP_LIKE(CPF, '[0-9]{3}\.[0-9]{3}\.[0-9]{3}\-[0-9]{2}'))
);

CREATE TABLE FESTAS (
    DATA_HORA DATE,
    LOCAL VARCHAR2(20),
    CLIENTE VARCHAR2(14) NOT NULL,
    TIPO VARCHAR2(9) NOT NULL,
    CASAMENTO_LOCAL_CERIMONIAL VARCHAR2(20),
    
    CONSTRAINT PK_FESTAS PRIMARY KEY (DATA_HORA, LOCAL),
    CONSTRAINT FK_FESTAS FOREIGN KEY (CLIENTE) REFERENCES CLIENTES(CPF) ON DELETE CASCADE,
    CONSTRAINT CHECK_TIPO CHECK(UPPER(TIPO) IN('CASAMENTO', 'INFANTIL'))
);

CREATE TABLE CRONOGRAMA (

    HORARIO DATE,
    
    CONSTRAINT PK_CRONOGRAMA PRIMARY KEY (DATA_HORA, LOCAL),
    CONSTRAINT FK_CRONOGRAMA FOREIGN KEY (DATA_HORA, LOCAL) REFERENCES FESTAS(DATA_HORA, LOCAL) ON DELETE CASCADE
);

CREATE TABLE JOGOS (
